<script src="https://cdn.jsdelivr.net/npm/@tensorflow/tfjs@latest/dist/tf.min.js"></script>
<script src="https://cdn.jsdelivr.net/npm/@teachablemachine/image@latest/dist/teachablemachine-image.min.js"></script>
<script type="text/javascript">
    // ลิงก์โมเดลเดิมของคุณ[cite: 1]
    const URL = "https://teachablemachine.withgoogle.com/models/ajE7BXwcJ/";[cite: 1]

    let model, webcam, labelContainer, maxPredictions;

    async function init() {
        const modelURL = URL + "model.json";[cite: 1]
        const metadataURL = URL + "metadata.json";[cite: 1]

        model = await tmImage.load(modelURL, metadataURL);[cite: 1]
        maxPredictions = model.getTotalClasses();[cite: 1]

        const flip = true; 
        webcam = new tmImage.Webcam(200, 200, flip);[cite: 1]
        await webcam.setup();[cite: 1]
        await webcam.play();[cite: 1]
        window.requestAnimationFrame(loop);[cite: 1]

        // แสดงผลกล้อง
        document.getElementById("webcam-container").innerHTML = "";
        document.getElementById("webcam-container").appendChild(webcam.canvas);[cite: 1]
        
        // สร้างโครงสร้างแถบผลลัพธ์ล่วงหน้า (ทำครั้งเดียวเพื่อไม่ให้หน้าเว็บกระตุก)
        labelContainer = document.getElementById("label-container");[cite: 1]
        labelContainer.innerHTML = ""; 
        for (let i = 0; i < maxPredictions; i++) {
            const resultItem = document.createElement("div");
            resultItem.className = "result-item";
            resultItem.innerHTML = `
                <span class="class-label" id="label-name-${i}">-</span>
                <div class="probability-bar-container">
                    <div class="probability-bar-fill" id="bar-fill-${i}"></div>
                </div>
                <span class="probability-text" id="prob-text-${i}">0%</span>
            `;
            labelContainer.appendChild(resultItem);
        }
    }

    async function loop() {
        webcam.update();[cite: 1]
        await predict();[cite: 1]
        window.requestAnimationFrame(loop);[cite: 1]
    }

    async function predict() {
        const prediction = await model.predict(webcam.canvas);[cite: 1]
        for (let i = 0; i < maxPredictions; i++) {
            const className = prediction[i].className;[cite: 1]
            const percentage = (prediction[i].probability * 100).toFixed(0);
            
            // อัปเดตข้อมูลลงในแถบที่สร้างไว้ล่วงหน้า (ห้ามลบแล้วสร้างใหม่ทุกเฟรม หน้าเว็บจะค้าง)
            if(document.getElementById(`label-name-${i}`)) {
                document.getElementById(`label-name-${i}`).innerText = className;
                document.getElementById(`bar-fill-${i}`).style.width = percentage + "%";
                document.getElementById(`prob-text-${i}`).innerText = percentage + "%";
            }
        }
    }
</script>
